% ── Page Layout ──────────────────────────────────────────────────────
% 17 x 24 cm: standard Dutch dissertation format
\usepackage[
  paperwidth=170mm, paperheight=240mm,
  top=2.5cm, bottom=2.5cm,
  inner=2.8cm, outer=1.8cm,
  headsep=0.6cm
]{geometry}

% ── Line spacing
\usepackage{setspace}
\setstretch{1.2}

% ── Paragraph spacing ────────────────────────────────────────────────
\setlength{\parindent}{0em}
\setlength{\parskip}{4pt plus 3pt minus 2pt}

% ── Caption style ─────────────────────────────────────────────────────
\usepackage{caption}
\DeclareCaptionFont{lining}{\addfontfeatures{Numbers=Lining}}
\captionsetup[figure]{
  labelfont={bf, lining},
  textfont={lining},
  font=normalsize,
  labelsep=period
}
\captionsetup[table]{
  labelfont={bf, lining},
  textfont={lining},
  font=normalsize,
  labelsep=period
}
\counterwithin{figure}{chapter}
\counterwithin{table}{chapter}

% ── Section heading style ─────────────────────────────────────────────
\usepackage{titlesec}

% Chapter: right-aligned title + large background chapter number
\renewcommand{\thechapter}{\arabic{chapter}}
\titleformat{\chapter}[display]
  {\Huge\bfseries\raggedleft}
  {\makebox[0pt][r]{\fontsize{120}{0}\selectfont\thechapter}}
  {12pt}
  {}
\titlespacing*{\chapter}{0pt}{-30pt}{28pt}

\titleformat{\section}
  {\headerfont\Large\bfseries}{\thesection}{0.5em}{}
\titleformat{\subsection}
  {\headerfont\large\bfseries}{\thesubsection}{0.5em}{}
\titleformat{\subsubsection}
  {\headerfont\normalsize\bfseries}{\thesubsubsection}{0.5em}{}
\titleformat{\paragraph}[runin]
  {\headerfont\normalsize\bfseries}{\theparagraph}{0.5em}{}

\titlespacing{\section}{0pt}{5pt}{0.5pt}
\titlespacing{\subsection}{0pt}{8pt}{1pt}
\titlespacing{\subsubsection}{0pt}{8pt}{1pt}
\titlespacing{\paragraph}{0pt}{0pt}{0.5em}

% ── Equation numbering per chapter ────────────────────────────────────
\numberwithin{equation}{chapter}

% ── Header & Footer ──────────────────────────────────────────────────
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[LE]{\headerfont\nouppercase{\leftmark}}
\fancyhead[RO]{\headerfont\nouppercase{\rightmark}}
\fancyhead[RE,LO]{\headerfont\thepage}
\renewcommand{\headrulewidth}{0.4pt}
\renewcommand{\chaptermark}[1]{\markboth{\thechapter.\ #1}{}}

% plain style: no header, centered page number at bottom
\fancypagestyle{plain}{
  \fancyhf{}
  \fancyfoot[C]{\headerfont\thepage}
  \renewcommand{\headrulewidth}{0pt}
  \renewcommand{\footrulewidth}{0pt}
}

% ── tcolorbox Styles ─────────────────────────────────────────────────
\definecolor{tudelft}{HTML}{00A6D6}

\newtcolorbox{keybox}[1][Key Result]{
  colback=tudelft!8,
  colframe=tudelft,
  coltitle=white,
  title=#1,
  fonttitle=\headerfont\bfseries,
  breakable
}

% ── Table of Contents depth ───────────────────────────────────────────
\setcounter{tocdepth}{1}
